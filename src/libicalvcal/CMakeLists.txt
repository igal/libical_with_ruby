)

set(icalvcal_LIB_DEVSRCS vcc.c)

SET(icalvcal_LIB_SRCS
   icalvcal.h
   icalvcal.c
   port.h
   vcc.h
   vobject.h
   vobject.c
   vcaltmp.h
   vcaltmp.c
   ${icalvcal_LIB_DEVSRCS}
)

    list(APPEND icalvcal_LIB_SRCS ../icalvcal.def)
